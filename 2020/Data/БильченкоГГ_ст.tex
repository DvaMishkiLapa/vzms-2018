\textbf{1. Дифференциальные уравнения

%\indent
    Рассматривается движение

%%% ==== \\

$\ell=const$,

$\omega=const$,

\phantom{0}

\,{;}
\eqno(1)

%%%%%%%%%%%%%%%%%%%%%%%%%

%%%%%%%%%%%%%%%%%%%%%%%%%

\beta \cdot
\left (
    \cos \varphi
    -f\cdot \sin \varphi
  \right )

	+\gamma
\phantom{0}

\mbox{при}

\dot{x}<0
\,{;}
\eqno(2)
\]
